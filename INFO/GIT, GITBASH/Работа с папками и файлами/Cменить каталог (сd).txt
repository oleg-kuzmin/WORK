# Change Directory (сменить каталог)

# Синтаксис:
cd имя_папки
cd имя_папки/имя_папки
# Если в названии папки есть пробелы, используйте кавычки:
cd "Рабочий стол"

# GitBash
cd Desktop # переходим на рабочий стол
cd ..      # вернуться в родительскую директорию (то есть на уровень выше)
cd ~       # переходим домашнюю директорию
